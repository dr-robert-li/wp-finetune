import yaml, sys
from pathlib import Path

root = Path("/home/robert_li/Desktop/projects/wp-finetune")

files_to_check = [
    root / "scripts/adaptive_planner.py",
    root / "config/adaptive_planning.yaml",
    root / ".claude/skills/wp-finetune:adaptive-planner/SKILL.md",
    root / ".claude/skills/wp-finetune:run-training/SKILL.md",
    root / ".claude/skills/wp-finetune:observe-training/SKILL.md",
]
for f in files_to_check:
    src = f.read_text()
    # Check for old thermal thresholds in thermal context
    for line_no, line in enumerate(src.split("\n"), 1):
        l = line.lower()
        if ("80c" in l or "80°c" in l) and ("temp" in l or "thermal" in l or "warn" in l):
            if "80w" not in l and "watts" not in l:
                errors.append(f"{f.name}:{line_no}: Old 80C thermal threshold found")
        if ("83c" in l or "83°c" in l):
            errors.append(f"{f.name}:{line_no}: Old 83C threshold found")

# 2. Correct API names
skill_src = (root / ".claude/skills/wp-finetune:adaptive-planner/SKILL.md").read_text()
planner_src = (root / "scripts/adaptive_planner.py").read_text()
train_src = (root / "scripts/train_model.py").read_text()

wrong_apis = {
    "compute_effective_scale": "Use 'compute' from telemetry.effective_scale",
    "run_probe": "Use 'prepare_probe' and 'evaluate_probe'",
    "record_run": "AnchorStore has no record_run method",
    "has_anchor_for": "AnchorStore has no has_anchor_for method",
    "last_safe_config": "AnchorStore has no last_safe_config method",
    "probe_in_cooldown": "AnchorStore has no probe_in_cooldown method",
}
for wrong, fix in wrong_apis.items():
    for name, src in [("skill", skill_src), ("planner", planner_src), ("train", train_src)]:
        if wrong in src:
            errors.append(f"{name}: Wrong API '{wrong}' -- {fix}")

# 3. Forbidden patterns
if "drop_caches" in train_src:
    errors.append("train_model.py: FORBIDDEN sudo drop_caches")
if "builtins.print" in train_src:
    errors.append("train_model.py: FORBIDDEN builtins.print monkey-patch")

# 4. Required patterns
checks = {
    "planner has classify_power_zone": "classify_power_zone" in planner_src,
    "planner has couple_batch_grad_accum": "couple_batch_grad_accum" in planner_src,
    "planner has apply_ladder": "apply_ladder" in planner_src,
    "planner has compute_batch_ceiling": "compute_batch_ceiling" in planner_src,
    "planner uses round()": "round(" in planner_src,
    "planner has max_drift": "max_drift" in planner_src,
    "train has GPUSampler": "GPUSampler" in train_src,
    "train has _unsloth_actuals": "_unsloth_actuals" in train_src,
    "train has trainer.args": "trainer.args" in train_src,
    "train has classify_failure": "classify_failure" in train_src,
    "skill has adaptive_planner ref": "adaptive_planner" in skill_src,
    "skill has prepare_probe": "prepare_probe" in skill_src,
    "skill has evaluate_probe": "evaluate_probe" in skill_src,
    "skill has AnchorStore": "AnchorStore" in skill_src,
}
for desc, passed in checks.items():
    if not passed:
        errors.append(f"Missing: {desc}")

# 5. YAML validity
try:
    cfg = yaml.safe_load((root / "config/adaptive_planning.yaml").read_text())
    assert cfg["thermal_brake"]["warning_temp"] == 82
    assert cfg["thermal_brake"]["emergency_temp"] == 85
    assert cfg["probe"]["steps"] == 5
    assert cfg["coupling"]["max_drift"] == 1
except Exception as e:
    errors.append(f"adaptive_planning.yaml: {e}")

# 6. dgx_toolbox.yaml mount
try:
    dgx = yaml.safe_load((root / "config/dgx_toolbox.yaml").read_text())
    if "dgx_telemetry" not in dgx.get("extra_mounts", {}):
        errors.append("dgx_toolbox.yaml: Missing dgx_telemetry mount")
    if "container_env" not in dgx:
        errors.append("dgx_toolbox.yaml: Missing container_env section")
except Exception as e:
    errors.append(f"dgx_toolbox.yaml: {e}")

# 7. Unit tests pass
import subprocess
result = subprocess.run(
    [sys.executable, "-m", "pytest", "tests/test_adaptive_planner.py", "-x", "-q"],
    capture_output=True, text=True, cwd=str(root)
)
if result.returncode != 0:
    errors.append(f"Unit tests failed: {result.stdout[-500:]}")

# Report
print("=" * 60)
print("CROSS-FILE VERIFICATION REPORT")
print("=" * 60)
if errors:
    print(f"\nERRORS ({len(errors)}):")
    for e in errors:
        print(f"  FAIL: {e}")
if warnings:
    print(f"\nWARNINGS ({len(warnings)}):")
    for w in warnings:
        print(f"  WARN: {w}")
if not errors:
    print("\nAll checks PASSED")
print("=" * 60)
sys.exit(1 if errors else 0)
