```python
"""Merge reasoning batches from Claude Code agents into final bulk JSON.

Each agent writes data/phase4_reasoning/{stream}/batches/batch_{N}.json
This script reads all batches, applies quality gates, and writes the merged bulk file.
"""
import json, sys, argparse
from pathlib import Path
from collections import Counter

PROJECT_ROOT = Path(__file__).resolve().parent.parent

def merge_cot_batches():
    from scripts.generate_deep_judge_cot import passes_quality_gate, verify_citation_accuracy, REQUIRED_DIMENSIONS

    batches_dir = PROJECT_ROOT / "data" / "phase4_reasoning" / "deep_judge_cot" / "batches"
    output_path = PROJECT_ROOT / "data" / "phase4_reasoning" / "deep_judge_cot" / "deep_judge_cot_bulk.json"

    all_examples = []
    rejected_quality = 0
    rejected_citation = 0
    parse_failures = 0

    for batch_file in sorted(batches_dir.glob("batch_*.json")):
        try:
            batch = json.loads(batch_file.read_text())
        except json.JSONDecodeError:
            parse_failures += 1
            continue
        for ex in batch:
            result = ex.get("reasoning", {})
            source_code = ex.get("code", "")
            if not passes_quality_gate(result, source_code):
                rejected_quality += 1
                continue
            citation_meta = verify_citation_accuracy(result, source_code)
            if citation_meta["hallucination_ratio"] > 0.5:
                rejected_citation += 1
                continue
            ex["citation_accuracy"] = citation_meta
            all_examples.append(ex)

    output_path.parent.mkdir(parents=True, exist_ok=True)
    output_path.write_text(json.dumps(all_examples, indent=2))
    total_attempted = len(all_examples) + rejected_quality + rejected_citation + parse_failures
    print(f"Deep judge CoT merge: {len(all_examples)} accepted / {total_attempted} attempted")
    print(f"  Rejected (quality gate): {rejected_quality}")
    print(f"  Rejected (citation hallucination): {rejected_citation}")
    print(f"  Parse failures: {parse_failures}")
    print(f"  Parse failure rate: {parse_failures/total_attempted*100:.1f}%" if total_attempted else "")
    return len(all_examples)

def merge_ctf_batches():
    from scripts.generate_critique_then_fix import passes_quality_gate, php_lint_check, check_critique_fix_alignment, REQUIRED_DIMENSIONS

    batches_dir = PROJECT_ROOT / "data" / "phase4_reasoning" / "critique_then_fix" / "batches"
    output_path = PROJECT_ROOT / "data" / "phase4_reasoning" / "critique_then_fix" / "critique_then_fix_bulk.json"

    all_examples = []
    rejected_quality = 0
    rejected_lint = 0
    rejected_align = 0
    parse_failures = 0

    for batch_file in sorted(batches_dir.glob("batch_*.json")):
        try:
            batch = json.loads(batch_file.read_text())
        except json.JSONDecodeError:
            parse_failures += 1
            continue
        for ex in batch:
            result = ex.get("critique", {})
            corrected = ex.get("corrected_code", "")
            defective = ex.get("defective_code", "")
            if not passes_quality_gate(result, corrected):
                rejected_quality += 1
                continue
            lint_meta = php_lint_check(corrected)
            if not lint_meta.get("valid", True) and "php not available" not in lint_meta.get("errors", ""):
                rejected_lint += 1
                continue
            align_meta = check_critique_fix_alignment(result, defective, corrected)
            if align_meta["alignment_ratio"] < 0.3:
                rejected_align += 1
                continue
            ex["php_lint"] = lint_meta
            ex["critique_fix_alignment"] = align_meta
            all_examples.append(ex)

    output_path.parent.mkdir(parents=True, exist_ok=True)
    output_path.write_text(json.dumps(all_examples, indent=2))
    total_attempted = len(all_examples) + rejected_quality + rejected_lint + rejected_align + parse_failures
    print(f"Critique-then-fix merge: {len(all_examples)} accepted / {total_attempted} attempted")
    print(f"  Rejected (quality gate): {rejected_quality}")
    print(f"  Rejected (PHP lint): {rejected_lint}")
    print(f"  Rejected (alignment): {rejected_align}")
    print(f"  Parse failures: {parse_failures}")
    return len(all_examples)

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("stream", choices=["cot", "ctf", "both"])
    args = parser.parse_args()
    if args.stream in ("cot", "both"):
        merge_cot_batches()
    if args.stream in ("ctf", "both"):
        merge_ctf_batches()
```
