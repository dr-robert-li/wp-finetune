```bash
python scripts/generate_deep_judge_cot.py
```

This generates ~200 examples (default target is 50% of all Phase 1 functions). The script:
- Loads all Phase 1 passed + failed functions
- Samples 3 golden seeds per example as few-shot exemplars
- Applies the quality gate (all 9 dimensions, scores 1-10, verdict PASS/FAIL, citation accuracy check)
- Rejects examples where citation hallucination_ratio > 0.5
- Checkpoints every 20 examples
- Writes final output to `data/phase4_reasoning/deep_judge_cot/deep_judge_cot_bulk.json`
- Prints summary with total generated, parse failure rate, citation rejection rate, dimension coverage

After completion, run a comprehensive quality audit:
```bash
python -c "
import json
from pathlib import Path
bulk = json.load(open('data/phase4_reasoning/deep_judge_cot/deep_judge_cot_bulk.json'))
print(f'=== DEEP JUDGE CoT BULK AUDIT ===')
print(f'Total examples: {len(bulk)}')

# Verdict distribution
fails = sum(1 for ex in bulk if ex.get('reasoning', {}).get('verdict') == 'FAIL')
passes = sum(1 for ex in bulk if ex.get('reasoning', {}).get('verdict') == 'PASS')
print(f'Verdicts: {passes} PASS, {fails} FAIL')

# Dimension coverage
dims = set()
for ex in bulk:
    dims.update(ex.get('reasoning', {}).get('dimension_analysis', {}).keys())
print(f'Dimensions: {sorted(dims)}')

# Citation accuracy (aggregate)
ca_ratios = [ex.get('citation_accuracy', {}).get('hallucination_ratio', 0) for ex in bulk if 'citation_accuracy' in ex]
if ca_ratios:
    print(f'Citation hallucination: mean={sum(ca_ratios)/len(ca_ratios):.3f}, max={max(ca_ratios):.3f}, p90={sorted(ca_ratios)[int(len(ca_ratios)*0.9)]:.3f}')
    high_hallucination = sum(1 for r in ca_ratios if r > 0.3)
    print(f'Examples with hallucination > 30%: {high_hallucination}/{len(ca_ratios)}')

# Score distribution per dimension
from collections import defaultdict
dim_scores = defaultdict(list)
for ex in bulk:
    for d, v in ex.get('reasoning', {}).get('dimension_analysis', {}).items():
        dim_scores[d].append(v.get('score', 0))
print(f'\nPer-dimension score means:')
for d in sorted(dim_scores.keys()):
    scores = dim_scores[d]
    print(f'  {d}: mean={sum(scores)/len(scores):.1f}, min={min(scores)}, max={max(scores)}')

# Duplicate check (exact code match)
codes = [ex.get('code', '') for ex in bulk]
unique_codes = set(codes)
print(f'\nDuplicates: {len(codes) - len(unique_codes)} exact code duplicates out of {len(codes)}')

assert len(bulk) >= 100, f'Need >= 100 bulk CoT, got {len(bulk)}'
print('\nBulk CoT audit PASSED')
"
```
