```python
MIN_FREE_MEM_GIB = 60  # Phase 4.4: RAISE to 80 GiB per RESEARCH.md Pitfall 1

def _check_free_memory_gib() -> float:
    if not shutil.which("free"):
        return float("inf")
    out = subprocess.check_output(["free", "-b"], text=True)
    for line in out.splitlines():
        if line.startswith("Mem:"):
            parts = line.split()
            available_bytes = int(parts[6]) if len(parts) >= 7 else int(parts[3])
            return available_bytes / (1024 ** 3)
    return float("inf")

free_gib = _check_free_memory_gib()
print(f"[pre-flight] Available unified memory: {free_gib:.1f} GiB (floor: {MIN_FREE_MEM_GIB} GiB)")
if free_gib < MIN_FREE_MEM_GIB:
    print(f"[pre-flight] ABORT: {free_gib:.1f} GiB available < {MIN_FREE_MEM_GIB} GiB floor. ...")
    sys.exit(2)
```

**Streaming-log Popen pattern** (lines 85-97):
```python
log = PROJECT_ROOT / "logs" / "phase4.4" / f"wave0_smoke_{ts}.log"
log.parent.mkdir(parents=True, exist_ok=True)

# CRITICAL: line-buffered streaming so partial output survives a CUDA OOM /
# NVRM kill / SIGKILL. dgx.execute(capture=True) buffers and loses it.
with log.open("w", buffering=1) as f:
    f.write(f"=== CMD ===\n{' '.join(docker_cmd)}\n=== OUTPUT ===\n")
    f.flush()
    proc = subprocess.Popen(docker_cmd, stdout=f, stderr=subprocess.STDOUT)
    try:
        rc = proc.wait(timeout=3600)
    except subprocess.TimeoutExpired:
        proc.kill()
        rc = -1
    f.write(f"\n=== RETURNCODE ===\n{rc}\n")

if rc != 0:
    sys.exit(1)
```

```python
def _get_vllm_endpoint() -> str:
    try:
        from scripts.dgx_toolbox import get_toolbox
        dgx = get_toolbox()

    except Exception:

def _vllm_health_check(endpoint: str) -> bool:
    try:
        import urllib.request
        health_url = endpoint.rstrip("/v1").rstrip("/") + "/health"
        req = urllib.request.urlopen(health_url, timeout=5)
        return req.status == 200
    except Exception:
        return False

def _wait_for_vllm(endpoint: str, timeout_s: int = 900) -> bool:
    deadline = time.time() + timeout_s
    attempt = 0
    while time.time() < deadline:
        if _vllm_health_check(endpoint):
            logger.info(f"vLLM is healthy at {endpoint} (attempt {attempt + 1})")
            return True
        attempt += 1
        if attempt % 12 == 0 and _is_vllm_crash_looping():
            logger.error("vLLM container is crash-looping. Aborting health wait.")
            return False
        time.sleep(VLLM_HEALTH_POLL_S)
    return False
```

**Crash-loop detection** (`run_eval_triage.py:495-530`):
```python
def _is_vllm_crash_looping() -> bool:
    cname = _get_vllm_container_name()
    try:
        result = subprocess.run(
            ["docker", "inspect", "--format",
             "{{.State.Status}} {{.RestartCount}}", cname],
            capture_output=True, text=True, timeout=5,
        )
        if result.returncode == 0:
            parts = result.stdout.strip().split()
            if len(parts) >= 2:
                status, restart_count = parts[0], int(parts[1])
                if status == "restarting" or restart_count >= 1:
                    return True
        logs = subprocess.run(["docker", "logs", "--tail", "50", cname],
                              capture_output=True, text=True, timeout=10)
        if logs.returncode == 0:
            output = logs.stdout + logs.stderr
            fatal_patterns = ["peft_helper.validate_legal", "modules_to_save",
                              "CUDA out of memory"]
            for pattern in fatal_patterns:
                if pattern in output:
                    return True
    except Exception:
        pass
    return False
```

**PHP lint with `<?php` prefix fix** (`scripts/generate_critique_then_fix.py:159-216`):
```python
def php_lint_check(code: str) -> dict:
    """Run PHP syntax check on the given code.
    NOTE: `php -l` only inspects content inside a `<?php` block; anything
    before the opening tag is treated as literal HTML and silently passes.
    We MUST prefix `<?php` before linting.
    """
    tmp_path = None
    try:
        if not code.lstrip().startswith("<?"):
            code_to_lint = "<?php\n" + code
        else:
            code_to_lint = code
        with tempfile.NamedTemporaryFile(suffix=".php", delete=False, mode="w",
                                         encoding="utf-8") as tmp:
            tmp.write(code_to_lint)
            tmp_path = tmp.name
        result = subprocess.run(
            ["php", "-l", tmp_path],
            capture_output=True, text=True, timeout=10,
        )
        if result.returncode == 0:
            return {"valid": True, "errors": ""}
        else:
            return {"valid": False, "errors": result.stderr.strip() or result.stdout.strip()}
    except FileNotFoundError:
        return {"valid": True, "errors": "php not available - skipped"}
    except subprocess.TimeoutExpired:
        return {"valid": False, "errors": "php lint timeout"}
    finally:
        if tmp_path and os.path.exists(tmp_path):
            try: os.unlink(tmp_path)
            except OSError: pass
```

**PHPCS subprocess invocation** (`eval/rubric_scorer.py:197-232`):
```python
def run_phpcs(code: str, standard: str = "WordPress") -> dict:
    """Write code to temp file, run phpcs --standard={standard} --report=json."""
    with tempfile.NamedTemporaryFile(mode="w", suffix=".php", delete=False) as tmp:
        tmp.write(code)
        tmp_path = tmp.name
    try:
        proc = subprocess.run(
            ["phpcs", f"--standard={standard}", "--report=json", tmp_path],
            capture_output=True, text=True, timeout=30,
        )
        raw = proc.stdout.strip()
        if not raw:
            return {"_unavailable": True, "files": {}, "totals": {"errors": 0, "warnings": 0}}
        return json.loads(raw)
    except FileNotFoundError:
        return {"_unavailable": True, "files": {}, "totals": {"errors": 0, "warnings": 0}}
    except subprocess.TimeoutExpired:
        return {"_timeout": True, "files": {}, "totals": {"errors": 0, "warnings": 0}}
    except json.JSONDecodeError:
        return {"_parse_error": True, "files": {}, "totals": {"errors": 0, "warnings": 0}}
    finally:
        Path(tmp_path).unlink(missing_ok=True)
```

**Per-batch loop with flag-not-fail discipline** (analog: `scripts/validate_ctf_batch.py:74-120` — collect errors per example, classify; VALIDATION.md T6 requires "unparseable → flag (not fail)"):
```python
import json
from pathlib import Path

def revl06_fix_correctness(pairs_jsonl: str, out_jsonl: str) -> dict:
    """For each CtF sample, extract <corrected_code> → run php -l + PHPCS +
    security scanner. Flag unparseable; fail only on lint-syntax errors."""
    n_total = 0; n_lint_pass = 0; n_phpcs_pass = 0; n_sec_pass = 0; n_unparseable = 0
    with open(pairs_jsonl) as fin, open(out_jsonl, "w") as fout:
        for line in fin:
            ex = json.loads(line)
            n_total += 1
            response = ex.get("response", "")
            corrected = extract_corrected_code_from_xml(response)
            if corrected is None or len(corrected.strip()) < 20:
                n_unparseable += 1
                fout.write(json.dumps({
                    "example_idx": ex.get("example_idx"),
                    "status": "unparseable",
                }) + "\n")
                continue
            lint = php_lint_check(corrected)
            phpcs = run_phpcs(corrected, standard="WordPress")
            # security scanner: per RESEARCH.md, use rubric_scorer's existing
            # SEC-N04-suppressed path (already covers severity 4→2 fix)
            from eval.rubric_scorer import score_code
            rubric = score_code(corrected)
            sec_pass = rubric.dimension_scores.get("D2_security", 0) >= 8.0
            if lint["valid"]: n_lint_pass += 1
            if not phpcs.get("_unavailable") and phpcs.get("totals", {}).get("errors", 1) == 0:
                n_phpcs_pass += 1
            if sec_pass: n_sec_pass += 1
            fout.write(json.dumps({
                "example_idx": ex.get("example_idx"),
                "lint_valid": lint["valid"],
                "phpcs_errors": phpcs.get("totals", {}).get("errors"),
                "security_score": rubric.dimension_scores.get("D2_security"),
                "status": "ok",
            }) + "\n")
    return {
        "n_total": n_total, "n_unparseable": n_unparseable,
        "lint_pass_rate": n_lint_pass / max(n_total - n_unparseable, 1),
        "phpcs_pass_rate": n_phpcs_pass / max(n_total - n_unparseable, 1),
        "security_pass_rate": n_sec_pass / max(n_total - n_unparseable, 1),
```

**Pattern (RESEARCH.md Example 4):**
```python
import json
from sklearn.metrics import confusion_matrix, precision_recall_fscore_support

def revl07_confusion_matrix(
    eval_judge_pairs_jsonl: str,
    output_path: str,
    score_threshold: float = 7.0,
) -> dict:
    per_dim = {f"d{i}": {"y_true": [], "y_pred": []} for i in range(1, 10)}
    with open(eval_judge_pairs_jsonl) as f:
        for line in f:
            ex = json.loads(line)
            gt = ex["gt_scores"]; pred = ex["model_scores"]
            for d in per_dim:
                per_dim[d]["y_true"].append(int(gt[d] >= score_threshold))
                per_dim[d]["y_pred"].append(int(pred[d] >= score_threshold))

    result = {}
    for d, arrs in per_dim.items():
        tn, fp, fn, tp = confusion_matrix(
            arrs["y_true"], arrs["y_pred"], labels=[0, 1],
        ).ravel()
        p, r, f1, _ = precision_recall_fscore_support(
            arrs["y_true"], arrs["y_pred"],
            average="binary", zero_division=0,
        )
        result[d] = {"tp": int(tp), "tn": int(tn), "fp": int(fp), "fn": int(fn),
                     "precision": float(p), "recall": float(r), "f1": float(f1)}

    with open(output_path, "w") as f:
        json.dump({"score_threshold": score_threshold, "per_dimension": result},
                  f, indent=2)
    return result
```

```python
import json, re
from transformers import AutoTokenizer
import numpy as np

REASONING_RE = re.compile(r"\[REASONING\](.*?)\[/REASONING\]", re.DOTALL)

def revl08_length_distribution(
    eval_judge_pairs_jsonl: str,
    output_path: str,
    tokenizer_dir: str = "adapters/tokenizer",   # MUST match training tokenizer
    flag_p95_max: int = 6000,
    flag_median_min: int = 500,
) -> dict:

    lengths = []
    with open(eval_judge_pairs_jsonl) as f:

        "p95": int(np.percentile(arr, 95)),
        "max": int(np.max(arr)),
        "mean": float(np.mean(arr)),
        "flags": {
            "p95_too_high": bool(np.percentile(arr, 95) > flag_p95_max),
            "median_too_low": bool(np.median(arr) < flag_median_min),
        },
    }
    with open(output_path, "w") as f:
        json.dump(summary, f, indent=2)
    return summary
```

```python
def stratified_pick(samples: list, n_high: int = 4, n_low: int = 4, n_rand: int = 2):
    # high: top-N by overall score; low: closest-to-boundary 7.0; random: rest
    sorted_by_overall = sorted(samples, key=lambda s: s["model_scores"]["overall"],
                               reverse=True)
    high = sorted_by_overall[:n_high]
    boundary = sorted(samples,
                      key=lambda s: abs(s["model_scores"]["overall"] - 7.0))[:n_low]
    remaining = [s for s in samples if s not in high and s not in boundary]
    random.seed(42)
    rand = random.sample(remaining, min(n_rand, len(remaining)))
    return high + boundary + rand
```
