def _wilson_upper(k, n, z=1.96):
    if n == 0:
        return float("nan")
    p = k / n
    denom = 1 + z * z / n
    center = (p + z * z / (2 * n)) / denom
    half = (z * math.sqrt(p * (1 - p) / n + z * z / (4 * n * n))) / denom
    return center + half
# Gate (line 276): arm_pass = (rate <= 0.10) and (wu <= 0.15)

def _codes(path):
    """Extract user-message content set from a JSONL for leakage comparison."""
    out = set()
    for l in open(path):
        l = l.strip()
        if not l:
            continue
        r = json.loads(l)
        u = next((m["content"] for m in r["messages"] if m["role"] == "user"), "")
        out.add(u)
    return out

# In main(), after sampling extra_rows:
extra_codes = {r["messages"][0]["content"] for r in extra_rows}
leak_sent = extra_codes & _codes(args.sentinel)
leak_val  = extra_codes & _codes(args.val)
if leak_sent:
    print(f"LEAKAGE: {len(leak_sent)} replay rows share code with sentinel", file=sys.stderr)
    return 1
if leak_val:
    print(f"LEAKAGE: {len(leak_val)} replay rows share code with val", file=sys.stderr)
    return 1

### Manifest incremental write
**Source:** `scripts/tinker_reasoning_sft.py` lines 74–78, 220–221  
**Apply to:** any script that launches a Tinker training run  
```python
def _write_manifest(path, payload):
    os.makedirs(os.path.dirname(path), exist_ok=True)
    with open(path, "w") as f:
        json.dump(payload, f, indent=2)
    print(f"[manifest] wrote {path}", flush=True)
```
Write after EVERY epoch, not only at end. Prevents stranded checkpoints if the process dies.

### Leakage guard
**Source:** `scripts/build_augmented_train.py` lines 24–56  
**Apply to:** `build_replay_mix.py` (new replay rows from phase1 pool)  
Any new rows drawn from `data/phase1_extraction/output/passed/` must be checked against both `invalid_php_sentinel.jsonl` and `openai_val.jsonl` before the file is written.

**Source:** `scripts/run_eval_reasoning.py` lines 351–400  
**Apply to:** `run_grid_eval.py` per-candidate result accumulation
